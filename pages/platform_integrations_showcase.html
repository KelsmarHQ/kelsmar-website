Quick Start Example


// Initialize Kelsmar API client
const kelsmar = new KelsmarAPI({
  apiKey: 'your-api-key',
  environment: 'production'
});

// Create a new campaign
const campaign = await kelsmar.campaigns.create({
  name: 'Welcome Series',
  type: 'email',
  audience: 'new-subscribers',
  aiGenerate: true
});

// Sync with external platform
await kelsmar.integrations.sync({
  platform: 'klaviyo',
  campaignId: campaign.id
});
